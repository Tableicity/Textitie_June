import os
import time
import json
import requests
from datetime import datetime

# ==========================================
# CONFIGURATION & CONSTANTS
# ==========================================
TEXTLINE_API_BASE = "https://textline.com"
TEXTLINE_TOKEN = os.environ.get("TEXTLINE_API_TOKEN") # Stored safely in Replit Secrets

HEADERS = {
    "X-TFA-Token": TEXTLINE_TOKEN,
    "Accept": "application/json",
    "Content-Type": "application/json"
}

# Local scratch directory inside your Replit workspace for staging raw data
STAGING_DIR = "./textline_dump_staging"
os.makedirs(STAGING_DIR, exist_ok=True)

# ==========================================
# RESILIENT REQUEST HANDLER (Handles 429s)
# ==========================================
def safe_request(url, params=None):
    """Executes API requests with automated exponential backoff for rate limits."""
    backoff = 2
    while True:
        response = requests.get(url, headers=HEADERS, params=params)
        
        if response.status_code == 200:
            return response.json()
            
        elif response.status_code == 429:
            # Handle rate limiting dynamically using Textline's Retry-After header
            retry_after = int(response.headers.get("Retry-After", backoff))
            print(f"⚠️ [429 Too Many Requests] Slammed rate limit. Sleeping for {retry_after}s...")
            time.sleep(retry_after)
            backoff *= 2 # Fallback backoff scale multiplier
            
        elif response.status_code == 401:
            raise Exception("❌ Invalid Textline API Token. Check configuration.")
            
        else:
            print(f"❌ HTTP Error {response.status_code}: {response.text}")
            return None

# ==========================================
# CONVERSATION HARVESTER LOOP
# ==========================================
def harvest_all_conversations():
    """Paginates recursively over the conversation ledger and drops logs into local storage."""
    print("🚀 Initializing Textline Scroll & Dump Engine...")
    
    endpoint = f"{TEXTLINE_API_BASE}/conversations.json"
    current_page = 1
    next_uuid_cursor = None  # Textline uses pagination tokens/cursors
    
    while True:
        print(f"📦 Fetching Conversation Matrix - Page {current_page}...")
        
        params = {"limit": 50}
        if next_uuid_cursor:
            params["after"] = next_uuid_cursor # Token payload shifting pointer

        data = safe_request(endpoint, params=params)
        if not data or "conversations" not in data:
            print("🏁 No data returned or extraction complete.")
            break
            
        conversations = data["conversations"]
        if not conversations:
            print("🏁 Reached the bottom of the ledger matrix.")
            break

        # Dump this exact page frame directly to local workspace storage
        timestamp = datetime.utcnow().strftime("%Y%m%d_%H%M%S")
        file_path = f"{STAGING_DIR}/convos_page_{current_page}_{timestamp}.json"
        
        with open(file_path, "w") as f:
            json.dump(conversations, f, indent=4)
        print(f"💾 Staged {len(conversations)} conversations sequentially to {file_path}")

        # Drill into individual message loops for every conversation fetched on this page
        for convo in conversations:
            convo_id = convo.get("uuid")
            harvest_messages_for_conversation(convo_id)

        # Evaluate token parameters to decide if loop needs to spin forward or terminate
        # Textline formats pagination markers inside metadata blocks
        meta = data.get("meta", {})
        next_uuid_cursor = meta.get("next_page_token") or meta.get("after")
        
        if not next_uuid_cursor:
            print("🏁 Pagination termination token reached.")
            break
            
        current_page += 1
        time.sleep(0.5) # Gentle cooldown spacing to stay well below the 429 threshold

# ==========================================
# INNER LOOP: MESSAGE THREAD HARVESTER
# ==========================================
def harvest_messages_for_conversation(convo_id):
    """Paginates cleanly through an individual conversation to scrape full 3-year historical contexts."""
    print(f"🔍 Digging through timeline logs for Convo UUID: {convo_id}")
    endpoint = f"{TEXTLINE_API_BASE}/conversations/{convo_id}/messages.json"
    
    message_page = 1
    next_msg_cursor = None
    all_convo_messages = []

    while True:
        params = {"limit": 100}
        if next_msg_cursor:
            params["after"] = next_msg_cursor

        msg_data = safe_request(endpoint, params=params)
        if not msg_data or "messages" not in msg_data:
            break

        messages = msg_data["messages"]
        if not messages:
            break

        all_convo_messages.extend(messages)
        
        meta = msg_data.get("meta", {})
        next_msg_cursor = meta.get("next_page_token") or meta.get("after")
        
        if not next_msg_cursor:
            break
            
        message_page += 1
        time.sleep(0.2)

    # Save the consolidated conversation history as an isolated JSON file
    if all_convo_messages:
        msg_file_path = f"{STAGING_DIR}/messages_convo_{convo_id}.json"
        with open(msg_file_path, "w") as f:
            json.dump(all_convo_messages, f, indent=4)
        print(f"✅ Successfully staged {len(all_convo_messages)} historic text lines for thread {convo_id}")

# ==========================================
# MAIN EXECUTION VECTOR
# ==========================================
if __name__ == "__main__":
    if not TEXTLINE_TOKEN:
        print("❌ Error: TEXTLINE_API_TOKEN environment variable is not defined.")
    else:
        start_time = time.time()
        harvest_all_conversations()
        print(f"\n🎉 Extraction Cycle Complete. Total execution duration: {round(time.time() - start_time, 2)} seconds.")
        print(f"👉 All raw payload datasets successfully locked down inside '{STAGING_DIR}/'. Ready for the LLM Team.")

The Verification Layer Engine (verify_and_clean.py)
python
import os
import json
import glob
from datetime import datetime

# ==========================================
# CONFIGURATION
# ==========================================
STAGING_DIR = "./textline_dump_staging"
CLEANED_DIR = "./texitie_production_ready"
os.makedirs(CLEANED_DIR, exist_ok=True)

# ==========================================
# UTILITY: TIMESTAMP PARSER
# ==========================================
def parse_timestamp(ts_string):
    """Parses standard ISO/Textline timestamps into Python datetime objects."""
    if not ts_string:
        return None
    # Strips timezone offset if present for unified parsing
    clean_ts = ts_string.split("+")[0].split("Z")[0]
    try:
        return datetime.strptime(clean_ts, "%Y-%m-%dT%H:%M:%S.%f")
    except ValueError:
        try:
            return datetime.strptime(clean_ts, "%Y-%m-%dT%H:%M:%S")
        except ValueError:
            print(f"⚠️ Failed to parse timestamp format: {ts_string}")
            return None

# ==========================================
# CORE VERIFICATION ENGINE
# ==========================================
def process_staged_history():
    """Reads raw staged message logs, deduplicates payloads, and audits timeline integrity."""
    print("🧹 Initializing Texitie Data Verification Layer...")
    
    # Locate all staged conversation message files
    search_path = os.path.join(STAGING_DIR, "messages_convo_*.json")
    message_files = glob.glob(search_path)
    
    if not message_files:
        print("❌ No staged message profiles located. Run the Harvester script first.")
        return

    print(f"🔍 Found {len(message_files)} raw conversation logs to verify.\n")

    for file_path in message_files:
        convo_id = os.path.basename(file_path).replace("messages_convo_", "").replace(".json", "")
        
        with open(file_path, "r") as f:
            try:
                raw_messages = json.load(f)
            except json.JSONDecodeError:
                print(f"❌ Corrupted JSON file bypassed: {file_path}")
                continue

        # Setup runtime tracking dictionaries for this conversation
        unique_messages = {}
        duplicates_caught = 0
        timeline_breaks = 0
        
        # 1. Deduplication Phase
        for msg in raw_messages:
            msg_uuid = msg.get("uuid")
            if not msg_uuid:
                continue
                
            if msg_uuid in unique_messages:
                duplicates_caught += 1
            else:
                unique_messages[msg_uuid] = msg

        # 2. Timeline Ordering & Gap Auditing Phase
        # Sort messages chronologically by actual creation timestamp
        sorted_messages = sorted(
            unique_messages.values(),
            key=lambda x: parse_timestamp(x.get("created_at")) or datetime.min
        )

        # Audit timeline continuity to flag anomalies (e.g., massive anomalies or backward dates)
        previous_time = None
        for msg in sorted_messages:
            current_time = parse_timestamp(msg.get("created_at"))
            
            if previous_time and current_time:
                # Flag if chronology is structurally broken or corrupted
                if current_time < previous_time:
                    timeline_breaks += 1
                    
            previous_time = current_time

        # 3. Export Verified & Structured Payload
        output_payload = {
            "conversation_uuid": convo_id,
            "total_raw_processed": len(raw_messages),
            "clean_count": len(sorted_messages),
            "duplicates_removed": duplicates_caught,
            "chronology_anomalies": timeline_breaks,
            "verified_at": datetime.utcnow().isoformat(),
            "messages": sorted_messages
        }

        # Write clean data file to production-ready directory
        output_file_path = os.path.join(CLEANED_DIR, f"verified_convo_{convo_id}.json")
        with open(output_file_path, "w") as out_f:
            json.dump(output_payload, out_f, indent=4)

        # Print distinct per-conversation health metrics
        print(f"📊 Thread Audit [{convo_id}]:")
        print(f"   • Raw: {len(raw_messages)} lines ➔ Cleaned: {len(sorted_messages)} lines")
        print(f"   • Duplicates Dead-Dropped: {duplicates_caught}")
        print(f"   • Timeline Integrity Faults: {timeline_breaks}")
        print(f"   💾 Saved verified data matrix to: {output_file_path}\n")

    print("🎉 Verification complete. Your data pipeline is officially 100% clean.")

if __name__ == "__main__":
    process_staged_history()
